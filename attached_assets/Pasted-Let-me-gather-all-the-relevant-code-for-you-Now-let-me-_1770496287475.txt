### ROI Calculator Logic

```typescript
const WEEKS_PER_MONTH = 4;
const PODBRIEF_COST = 19; // $/month
const SUMMARY_DURATION = 7.5; // minutes - based on real average

// Default slider values
const [podcastLength, setPodcastLength] = useState(60);     // min:15, max:180, step:5
const [podcastsPerWeek, setPodcastsPerWeek] = useState(5);  // min:1,  max:20,  step:1
const [hourlyRate, setHourlyRate] = useState(30);            // min:10, max:200, step:5

// Calculator formula
const timeSavedPerPodcast = podcastLength - SUMMARY_DURATION;
const totalMinutesSaved = timeSavedPerPodcast * podcastsPerWeek * WEEKS_PER_MONTH;
const hoursSaved = totalMinutesSaved / 60;           // e.g. "14.0"
const grossValue = Math.round(hoursSaved * hourlyRate); // e.g. "$420"
const netSavings = Math.max(0, Math.round(grossValue - PODBRIEF_COST)); // e.g. "$401"
const roiMultiplier = Math.round(grossValue / PODBRIEF_COST); // e.g. "22x"
```
